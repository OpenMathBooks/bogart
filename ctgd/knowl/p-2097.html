\begin{align*} \sum_{n=2}^\infty d_n\frac{x^{n-1}}{(n-1)!} =\amp \sum_{n=2}^\infty d_{n-1}\frac{x^{n-1}}{(n-2)!} +\sum_{n=2}^\infty d_{n-2}\frac{x^{n-1}}{(n-2)!}\\ \sum_{n=1}^\infty d_n\frac{x^{n-1}}{(n-1)!} -d_1 =\amp x\sum_{n=2}^\infty d_{n-1}\frac{x^{n-2}}{(n-2)!} +xD(x)\\ D'(x) -d_1 =\amp xD'(x) +xD(x)\\ D'(x)(1-x) =\amp xD(x)\\ \frac{D'(x)}{D(x)} =\amp \frac{x}{1-x} \end{align*}

This gives us \(\ln D(x) = -\ln(1-x) -x +c\text{,}\) so that \(D(x) = \frac{1}{1-x}e^{-x}e^c\text{.}\) Since \(d_0=1\text{,}\) we have \(d(0)=1\text{,}\) so \(c=0\) and

\begin{align*} D(x) =\amp \frac{e^{-x}}{1-x}\\ =\amp \sum_{i=0}^\infty(-1)^i\frac{x^i}{i!}\cdot\sum_{j=0}^\infty x^j=\sum_{i=0}^\infty\left(\sum_{j=0}^i \frac{(-1)^j}{j!}\right) x^i. \end{align*}

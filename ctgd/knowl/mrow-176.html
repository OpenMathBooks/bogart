\begin{align*} a_{n-1}+a_{n-2} =\amp \sum_{k=0}^{n-1} \binom{n-1-k}{k} +\sum_{k=0}^{n-2} \binom{n-2-k}{k}\\ =\amp \sum_{k=0}^{n-1} \binom{n-1-k}{k}+\sum_{j=1}^{n-1}\binom{n-1-j}{j-1}\\ =\amp 1+\sum_{k=1}^{n-1}\binom{n-1-k}{k}+\binom{n-1-k}{k-1}\\ =\amp 1+\sum_{k=1}^{n-1}\binom{n-k}{k} = \sum_{k=0}^{n-1}\binom{n-k}{k} = a_n \end{align*} in-context

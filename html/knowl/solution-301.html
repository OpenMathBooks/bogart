\begin{align*} E_P(S_1\times S_2) =\amp \sum_{(x_1,x_2)\in S_1\times S_2} P(x_1)P(x_2)\\ =\amp \sum_{x_1:x_1\in S_1}\sum_{x_2:x_2\in S_2} P(x_1)P(x_2)\\ =\amp \sum_{x_1\in S_1}P(x_1)\sum_{x_2\in S_2}P(x_2)\\ =\amp \sum_{x_1\in S_1} P(x_1) E_{P_2}(S_2)\\ =\amp \left(\sum_{x_1\in S_1} P(x_1)\right)E_{P_2}(S_2)\\ =\amp E_{P_1}(S_1)E_{P_2}(S_2) \end{align*} in-context
